\squarebox[LightBlue]{
\chapterabstract{This chapter introduces a hybrid P4/NFV architecture combining hardware-based feature extraction using Tofino switches with an unsupervised machine learning model to improve the generalization of CG traffic detection. The proposed solution enhances processing rates while maintaining high identification accuracy through model adaptations addressing P4 limitations.}
